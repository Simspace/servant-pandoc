\section{\texorpdfstring{GET
\texttt{/hello;lang=\textless{}value\textgreater{}/:name}}{GET /hello;lang=\textless{}value\textgreater{}/:name}}

\subsection{Captures}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \emph{name}: name of the person to greet
\end{itemize}

\subsection{GET Parameters}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  capital

  \begin{description}
  \itemsep1pt\parskip0pt\parsep0pt
  \item[\textbf{Values}]
  \emph{true, false}
  \item[\textbf{Description}]
  Get the greeting message in uppercase (true) or not (false).Default is false.
  \end{description}
\end{itemize}

\subsection{Response}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  Status code 200
\item
  If you use ?capital=true (application/json)

\begin{Shaded}
\begin{Highlighting}[]
\StringTok{"HELLO, HASKELLER"}
\end{Highlighting}
\end{Shaded}
\item
  If you use ?capital=true (text/plain;charset=utf-8)

\begin{verbatim}
"HELLO, HASKELLER"
\end{verbatim}
\item
  If you use ?capital=false (application/json)

\begin{Shaded}
\begin{Highlighting}[]
\StringTok{"Hello, haskeller"}
\end{Highlighting}
\end{Shaded}
\item
  If you use ?capital=false (text/plain;charset=utf-8)

\begin{verbatim}
"Hello, haskeller"
\end{verbatim}
\end{itemize}
